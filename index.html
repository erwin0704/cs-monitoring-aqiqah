// ========================================== // BACKEND API - AQIQAH EXPRESS CS MONITORING // ========================================== const SS_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doPost(e) { const ss = SpreadsheetApp.openById(SS_ID); const action = e.parameter.action; // 1. API LOGIN if (action === 'login') { const sheetUser = ss.getSheetByName("Data_User"); const data = sheetUser.getDataRange().getValues(); // Hash PIN yang diinput untuk dicocokkan let hashedInputPin = hashString(e.parameter.pin); for (let i = 1; i < data.length; i++) { if (data[i][0] == e.parameter.id_cs && data[i][6] == "Aktif") { if (data[i][3] === hashedInputPin) { return createResponse({ status: "success", user: { id: data[i][0], nama: data[i][1], role: data[i][2], cabang: data[i][5] } }); } else { return createResponse({status: "error", message: "PIN Salah!"}); } } } return createResponse({status: "error", message: "ID Tidak Ditemukan atau Akun Tidak Aktif!"}); } // 1b. API SETUP PIN (Buat PIN Pertama Kali) if (action === 'setupPin') { const sheetUser = ss.getSheetByName("Data_User"); const data = sheetUser.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == e.parameter.id_cs) { if (data[i][3] && data[i][3] !== "") { return createResponse({status: "error", message: "PIN sudah pernah diatur. Gunakan fitur Ganti PIN atau minta Kode Akses baru."}); } if (data[i][4] != e.parameter.kode_akses) { return createResponse({status: "error", message: "Kode Akses salah atau sudah tidak berlaku!"}); } // Simpan Hash PIN & Kosongkan Kode Akses sheetUser.getRange(i + 1, 4).setValue(hashString(e.parameter.pin_baru)); sheetUser.getRange(i + 1, 5).setValue(""); // Hapus kode akses return createResponse({status: "success", message: "PIN berhasil dibuat! Silakan login."}); } } return createResponse({status: "error", message: "ID Tidak Ditemukan!"}); } // 1b2. API RESET PIN (Lupa PIN, pakai Kode Akses baru dari SPV) if (action === 'resetPin') { const sheetUser = ss.getSheetByName("Data_User"); const data = sheetUser.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == e.parameter.id_cs) { if (!data[i][4] || data[i][4] !== e.parameter.kode_akses) { return createResponse({status: "error", message: "Kode Akses salah atau sudah tidak berlaku!"}); } // Simpan Hash PIN baru & kosongkan Kode Akses (sekali pakai) sheetUser.getRange(i + 1, 4).setValue(hashString(e.parameter.pin_baru)); sheetUser.getRange(i + 1, 5).setValue(""); return createResponse({status: "success", message: "PIN berhasil direset! Silakan login dengan PIN baru."}); } } return createResponse({status: "error", message: "ID Tidak Ditemukan!"}); } // 1c. API CHANGE PIN (Ganti PIN Mandiri) if (action === 'changePin') { const sheetUser = ss.getSheetByName("Data_User"); const data = sheetUser.getDataRange().getValues(); let hashedOldPin = hashString(e.parameter.pin_lama); for (let i = 1; i < data.length; i++) { if (data[i][0] == e.parameter.id_cs) { if (data[i][3] !== hashedOldPin) { return createResponse({status: "error", message: "PIN Lama salah!"}); } sheetUser.getRange(i + 1, 4).setValue(hashString(e.parameter.pin_baru)); return createResponse({status: "success", message: "PIN berhasil diganti!"}); } } return createResponse({status: "error", message: "ID Tidak Ditemukan!"}); } // 2. API SUBMIT LAPORAN (Fase 3 sub-fitur 5: satu laporan per ID_CS+Tanggal. // Kalau ID_CS ini sudah pernah lapor di tanggal itu, submit ulang jadi UPDATE laporan yang sama, bukan bikin baris baru) if (action === 'submitLaporan') { const sheetLead = ss.getSheetByName("Data_Lead"); const sheetTrx = ss.getSheetByName("Data_Transaksi"); let idCs = e.parameter.id_cs; let tglLaporan = e.parameter.tanggal; let totalLead = e.parameter.total_lead; // CS Multi-Region: kalau form kirim breakdown per-region, Total_Lead dihitung ulang dari sum-nya // (bukan trust nilai total_lead dari client) supaya keduanya selalu konsisten. let leadPerRegionRaw = e.parameter.lead_per_region || ''; } } let dataInvoices; try { dataInvoices = JSON.parse(e.parameter.invoices); } catch (error) { return createResponse({status: "error", message: "Data invoice gagal diproses: " + error.message}); } let timestamp = Utilities.formatDate(new Date(), "GMT+7", "yyyy-MM-dd HH:mm:ss"); // Pisahkan baris Data_Transaksi milik laporan ID_CS+Tanggal ini (kandidat update/hapus) dari baris laporan lain // (dipakai buat cek duplikat No_Invoice lintas-laporan, supaya edit invoice sendiri tidak dianggap duplikat) const existingTrx = sheetTrx.getDataRange().getValues(); let rowsLaporanIni = []; // { rowNum (1-based di sheet), no_invoice, status } let existingInvoiceSet = new Set(); for (let i = 1; i < existingTrx.length; i++) { let noInv = String(existingTrx[i][2]).trim(); let tglLaporanRow = existingTrx[i][10] ? extractTanggal(existingTrx[i][10]) : extractTanggal(existingTrx[i][0]); if (existingTrx[i][1] === idCs && tglLaporanRow === tglLaporan) { rowsLaporanIni.push({ rowNum: i + 1, no_invoice: noInv, status: existingTrx[i][7] }); } else if (noInv && noInv !== '-') { existingInvoiceSet.add(noInv); } } let duplikat = []; let invoiceValid = dataInvoices.filter(inv => { let noInv = String(inv.no_invoice).trim(); if (noInv && noInv !== '-' && existingInvoiceSet.has(noInv)) { duplikat.push(noInv); 'Pending' : existingRow.status; sheetTrx.getRange(rowNum, 3, 1, 7).setValues([[inv.no_invoice, inv.nama_cust, inv.cabang, inv.qty, inv.revenue, statusBaru, inv.tgl_acara]]); if (statusBaru === 'Pending' && existingRow.status !== 'Pending') sheetTrx.getRange(rowNum, 10).setValue(""); } else { sheetTrx.appendRow([timestamp, idCs, inv.no_invoice, inv.nama_cust, inv.cabang, inv.qty, inv.revenue, "Pending", inv.tgl_acara, "", tglLaporan]); } }); // Baris lama yang tidak lagi ada di submit -> dihapus dari sheet (rowNum tertinggi dulu, aman krn semua update sudah selesai) rowsLaporanIni .filter(row => !noInvoiceSubmitted.has(row.no_invoice)) .map(row => row.rowNum) .sort((a, b) => b - a) .forEach(rowNum => sheetTrx.deleteRow(rowNum)); let msg = modeEdit ? "Perubahan laporan berhasil disimpan dan menunggu Review SPV." : "Data berhasil disimpan dan menunggu Review SPV."; if (duplikat.length > 0) { msg += " Perhatian: " + duplikat.length + " invoice dilewati karena No. Invoice sudah pernah tercatat sebelumnya."; } return createResponse({status: "success", message: msg, duplikat: duplikat}); } // 4. "Jabar,Jatim,Jogja") if (action === 'updateUserCabang') { const sheetUser = ss.getSheetByName("Data_User"); const data = sheetUser.getDataRange().getValues(); let idCs = e.parameter.id_cs; let cabang = String(e.parameter.cabang || '').trim(); if (!cabang) return createResponse({status: "error", message: "Pilih minimal satu Cabang/Region!"}); for (let i = 1; i < data.length; i++) { if (data[i][0] == idCs) { sheetUser.getRange(i + 1, 6).setValue(cabang); return createResponse({status: "success", message: "Cabang/Region " + idCs + " berhasil diperbarui!"}); } } return createResponse({status: "error", message: "ID Tidak Ditemukan!"}); } // 7.
